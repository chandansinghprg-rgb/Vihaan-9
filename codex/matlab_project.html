MATLAB — Loops & Conditionals
% For Loop — Sum of 1 to 10
total = 0;
for i = 1:10
    total = total + i;
end
disp(total)     % 55

% While Loop — Factorial of 5
n = 5; fact = 1; k = 1;
while k <= n
    fact = fact * k;
    k = k + 1;
end
disp(fact)      % 120

% Switch-Case
day = 3;
switch day
  case 1, disp('Monday')
  case 3, disp('Wednesday')
  otherwise, disp('Other')
end
total = 55
fact = 120
Wednesday

MATLAB — Polynomials
% Polynomial: p(x) = x² - 3x + 2
p = [1 -3 2];

% Evaluate at x = 4
val = polyval(p, 4)   % 4²-12+2 = 6

% Find Roots
r = roots(p)          % [2; 1]

% Derivative: dp/dx = 2x - 3
dp = polyder(p)       % [2 -3]

% Integral
ip = polyint(p)       % [1/3 -3/2 2 0]

% Curve Fitting (y = x²)
x = 1:5;
y = x.^2;
cf = polyfit(x, y, 2) % ≈ [1 0 0]
val = 6
r = [2; 1]
dp = [2 -3] → 2x - 3
ip = [0.3333 -1.5000 2 0]
cf ≈ [1.0000 0.0000 0.0000]
